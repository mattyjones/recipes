    \recipe{Black Bean and Sweet Potato Enchiladas}
    \ingred{
        \item Cilantro (chopped), 1/4 cup
        \item Sweet Potato (diced), 1
        \item Black Beans (cooked), 1 cup 
        \item Kidney Beans (cooked), 1 cup
        \item 8" Flour Tortilla
        \item Shredded Cheese, 2 1/2 cups
        \item Enchilida sauce, 6 cups
        \item Yellow Onion (diced), 1 large
        \item Yellow Squash (diced), 1 large
    }

    \instruct{
        \item Steam sweet potato until soft
        \item Cook squash until soft
        \item Mash sweet potatos and then fold in beans and squash
        \item Pour enough of the sauce to cover the bottom of a 9x13 baking dish
        \item Add mixture to flour tortilla and roll, then place tightly in baking dish
        \item When dish is packed as a single layer, add remaining sauce to cover enchiladas
        \item Add cheese and bake in the oven at 350 for 45 minutes
    }
