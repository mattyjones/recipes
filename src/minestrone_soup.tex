    \recipe{Minestrone Soup}
    \ingred{
        \item Olive Oil, 2 tbsp
        \item Yellow Onion (diced), 1 medium
        \item Garlic Clove (minced), 4
        \item Celery (diced), 2 stalks
        \item Carrot (diced), 1 large
        \item Green Beans, 1 1/2 cups
        \item Oragano, 1 tsp
        \item Basil, 1 tsp
        \item Salt, taste
        \item Pepper, taste 
        \item Diced Tomatos, (1) 28oz can
        \item Crushed Tomatos, (1) 14oz can
        \item Chicken Broth, 6 cups 
        \item Kidney Beans, (1) 15oz can
        \item Small Pasta Shape, 1 cup
    }

    \instruct{
        \item Sweat celery, onions, and carrots in a large soup pot until soft
        \item Drain and rinse beans, then add to soup pot
        \item Add remaining Ingredients except pasta to soup pan
        \item Bring pot to a boil and then reduce heat to a low simmer for 30m or until vegtables are fully cooked
        \item Serve in a bowl with pasta
    }
