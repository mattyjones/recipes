    \recipe{Enchilada Sauce}
    \ingred{
        \item Vegtable Stock, 2 1/4 cups
        \item Tomato Paste, 1/3 cup
        \item Mild Chili Powder, 2 1/2 tbsp
        \item Cummin, 2 tsp
        \item Oragano, 1 1/2 tsp
        \item Garlic Clove (minced), 3
        \item Soy Sauce, 2 tsp
        \item Lime Juice, 1 tbsp
        \item Chipolte Peppers in Adobo Sauce (minced), 1 tbsp
    }

    \instruct{
        \item Add everything to a meduim to a sauce pan except the lime juice
        \item Bring sauce to a boil, then reduce to a simmer to thick
        \item Add Lime Juice and remove from heat
    }
