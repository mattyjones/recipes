    \recipe{Barley and Sausage Casserole}
    \ingred{
        \item Bulk Sweet Italian Sausage, 4 oz
        \item Garlic Clove (minced), 2 Cloves
        \item Button mushrooms (sliced), 3 cup
        \item Yellow Onion (diced), 1/2 cup
        \item Barley (cooked), 2 cup
        \item Pasta Sauce, 1 1/2 cup
        \item Mozzarella Cheese, 6 oz
        \item Roasted Red Peppers, 1/2 cup
        \item Fresh Basil (chopped), 2 tbsp
    }

    \instruct{
        \item Add pork and garlic to the slow cooker
        \item Add salt, lime juice, cumin, cinnamon, and Cayenne to a small bowl
        \item Combine spice mixture, orange juice, and broth
        \item Pour broth mixture into slow cooker, use tongs to ensure pork is well covered
        \item Cook for 7 hours on low or until pork is falling apart.
        \item Remove pork and shred, add salt and pepper to taste
    }
