    \recipe{Sloppy Joes}
    \ingred{
        \item Yellow Onion (chopped), 1
        \item Bell Peppers (chopped), 2
        \item Ground Beef, 2 1/2 lbs
        \item Garlic Clove (minced), 5
        \item Light Brown Sugar, 2 tbsp
        \item Dry Mustard, 1 tsp
        \item Mild Chili Powder, 2 tsp
        \item Worchestire Sauce, to taste
    }

    \instruct{
        \item Brown ground beef and drain the fat and set aside
        \item Cook peppers and onions till onions are translucent then set aside
        \item Combine remaining ingredients, mix completely and bring to a boil
        \item Add beef, onions, and peppers to mixture and combine
        \item Reduce heat to a simmer and cook for 20m
        \item Serve on a Bulkie roll with loaded tater tots
    }
