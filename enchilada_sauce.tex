    \recipe{Enchilada Sauce}
    \ingred{
        \item Vegtable Oil, 1 tbsp
        \item Yellow Onion (fine chop), 1
        \item Chipolte Peppers in Adobo Sauce (minced), 1 tbsp
        \item Garlic Clove (minced), 5
        \item Mild Chili Powder, 2 tbsp
        \item Cummin, 2 tsp
        \item Coriander, 1 tsp
        \item Tomato Sauce, 1 28oz can
        \item Chicken Stock, 2 cups
    }

    \instruct{
        \item Add oil and onion to heavy pan and cook till onion is tender
        \item Add garlic, chili powder, adobo sauce, cumin, and coriander and cook for one minute, stiring constantly
        \item Stir in tomato sauce and stock and bring to a boil and then simmer till slightly thickened
    }
