    \recipe{Black Bean and Corn Quiona}
    \ingred{
        \item Quinoa, 1/2 cup
        \item Veg Stock, 3 cups
        \item Olive Oil, 3 tbsp
        \item Cilantro (chopped), 3 tbsp
        \item Lime juice, 2 tbsp
        \item Jalapeno (chopped and seeded), 1
        \item Red Bell Pepper (chopped), 1
        \item Shallot (chopped fine), 1
        \item Garlic Clove (minced), 2
        \item Corn (thawed), 12oz
        \item Black Beans (raw), 16oz 
        \item Mild Chili Powder, 1 tsp
        \item Cummin, 1/2 tsp
    }

    \instruct{
        \item Cook and drain black beans following the package directions
        \item Cook Quiona in the veggie stock following the package directions
        \item Cook peppers, shallots and garlic until shallots are translucent
        \item Add beans, corn, and seasoning and cook till peppers are soft
        \item Combine corn salsa, quinoa, and cilantro to a bowl and fold together gently
    }
