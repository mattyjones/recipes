    \recipe{Cuban Pork W/ Sweet Potatoes}
    \ingred{
        \item Sweet Potato (cut into 1/2" pieces), 1 lb
        \item Pork Loin (cut into 1/2" pieces), 1 lb
        \item Diced Tomatoes, 1 14oz can
        \item Jalapano (seeded, small dice), 1
        \item Garlic Clove, 1
        \item Orange Juice, 1/4 cup
        \item Green Onion (chopped), 2
        \item Table Salt, 1/2 tsp
        \item Pepper, 1/4 tsp
        \item Cummin, 1/4 tsp
        \item Lime Juice, 2 tbsp
        \item Cilantro, 2 tbsp
    }

    \instruct{
        \item Place all ingredients into a slow cooker
        \item Cook on low for 7 hours
    }
