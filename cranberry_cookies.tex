\recipe{Cranberry Cookies}
\ingred{\begin{itemize}
	\item Vanilla Pudding Cake Mix, 1 Pkg
	\item Vegtable Oil, 1/2 Cup
    \item 2 Eggs
    \item Chopped Cranberries, 1 Cup
\end{itemize}}


\begin{enumerate}
    \item Mix all ingredients in a bowl
    \item Bake in the oven at 350 \textdegree for 9-11 minutes or until golden brown
\end{enumerate}
