    \recipe{Mashed Potatoes}
    \ingred{
        \item Medium Russet Potatoes (peeled, diced), 2
        \item Unslated Butter (melted), 4 tbsp
        \item Salt, 1/2 tsp
        \item White Pepper, 1/2 tsp
    }

    \instruct{
        \item Add potatos to a soup pot with 1 gallon of cold water
        \item Bring Potatoes to a boil and simmer till you can easy poke a fork through
        \item Mix butter, salt, and pepper together
        \item Drain potatoes and mash lightly with a fork
        \item Add butter mixture to potatoes and mix to desired consistency of mashed potatoes
    }
