    \recipe{Chicken Enchiladas}
    \ingred{
        \item Cilantro (chopped), 1/4 cup
        \item Shallot (chopped fine), 1
        \item Anaheim Chili Peppers (seeded, small dice), 4 
        \item Boneless Chicken Breasts, 3 lbs
        \item 6" Corn tortilla, 24
        \item Shredded Cheese, 2 1/2 cups
        \item Enchilida sauce, 6 cups
    }

    \instruct{
        \item Put sauce in heavy pan and submerge chicken in it, cover and simmer till chicken is cooked
        \item Remove pot from burner and transfer chicken to a plate and shred
        \item Cook Quiona in the veggie stock following the package directions
        \item Cook peppers, shallots and garlic until shallots are translucent
        \item Add beans, corn, and seasoning and cook till peppers are soft
        \item Combine corn salsa, quinoa, and cilantro to a bowl and fold together gently
    }
